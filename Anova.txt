# Example code: ANOVA using Decision Tree Classifier

import numpy as np
import pandas as pd

from sklearn.datasets import load_breast_cancer
from sklearn.model_selection import train_test_split
from sklearn.feature_selection import SelectKBest, f_classif
from sklearn.tree import DecisionTreeClassifier
from sklearn.metrics import accuracy_score, classification_report

# 1. Load Breast Cancer Dataset
data = load_breast_cancer()

X = pd.DataFrame(data.data, columns=data.feature_names)
y = pd.Series(data.target)

print("Original Dataset Shape:")
print("X =", X.shape)
print("y =", y.shape)

# 2. Split into Training and Testing Sets (70%-30%)
X_train, X_test, y_train, y_test = train_test_split(
    X, y,
    test_size=0.30,
    random_state=42,
    stratify=y
)

# Train using all features
clf_all_features = DecisionTreeClassifier(random_state=42)

clf_all_features.fit(X_train, y_train)

y_pred_all = clf_all_features.predict(X_test)

acc_all = accuracy_score(y_test, y_pred_all)

print("\nAccuracy using all features:")
print(f"{acc_all:.4f}")

# 3. Feature Selection using ANOVA
k_features = 5

anova_selector = SelectKBest(
    score_func=f_classif,
    k=k_features
)

# Fit on training data and transform both sets
X_train_selected = anova_selector.fit_transform(
    X_train, y_train
)

X_test_selected = anova_selector.transform(
    X_test
)

# Get selected feature names and scores
selected_indices = anova_selector.get_support(indices=True)

selected_feature_names = X.columns[selected_indices]

selected_scores = anova_selector.scores_[selected_indices]

print("\nSelected Features:")
for feature, score in zip(selected_feature_names,
                          selected_scores):
    print(f"{feature}: {score:.2f}")

# 4. Model Training with Selected Features
clf_selected_features = DecisionTreeClassifier(
    random_state=42
)

clf_selected_features.fit(
    X_train_selected,
    y_train
)

y_pred_selected = clf_selected_features.predict(
    X_test_selected
)

acc_selected = accuracy_score(
    y_test,
    y_pred_selected
)

print(f"\nAccuracy with Top {k_features} features:")
print(f"{acc_selected:.4f}")

print("\nClassification Report:")
print(classification_report(
    y_test,
    y_pred_selected
))
